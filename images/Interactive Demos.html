    def get_color_map(groupings, palette='husl'):
        rgb_values = sns.color_palette(palette, len(groupings))
        color_map = dict(zip(groupings, rgb_values))
        return color_map
    ​
    def plot_pog(pog_df, color_by='Brand', color_map=None , title=None, palette='husl'):
        sns.set(rc={"figure.figsize": (40, 10)})
        
        if color_map is None:
            color_by_metric = pog_df[color_by].unique()
            color_map = get_color_map(color_by_metric, palette)
            
        plt.scatter(x=pog_df.SLHorizontal+pog_df.HOffset, y=pog_df.SLVertical+pog_df.VOffset,
                    s=pog_df.HFacing*pog_df.VFacing*1000,
                    marker='s',
                    c=pog_df[color_by].map(color_map))
        
        if title is None:
            title = pog_df.PlanogramName.unique()[0]
        
        plt.title(title, fontsize=40)
        plt.show()
